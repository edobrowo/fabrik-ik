cmake_minimum_required(VERSION 3.10)

project(FABRIK_IK VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -Werror -Wall -Wpedantic -D_THREAD_SAFE")

include_directories(/opt/homebrew/include)

link_directories(/opt/homebrew/lib)
link_libraries(SDL2)

set(SRC_DIR src)
set(BIN_DIR ${CMAKE_BINARY_DIR})
set(EXEC main)

file(GLOB SRC_FILES "${SRC_DIR}/*.cpp")

add_executable(${EXEC} ${SRC_FILES})

set_target_properties(${EXEC} PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${BIN_DIR})
